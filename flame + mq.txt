#define BLYNK_TEMPLATE_ID "TMPxxxx"
#define BLYNK_TEMPLATE_NAME "FlameMQ"
#define BLYNK_AUTH_TOKEN "YourToken"

#include <ESP8266WiFi.h>
#include <BlynkSimpleEsp8266.h>

char ssid[] = "WiFi";
char pass[] = "Password";

#define flamePin D1
#define mqPin A0

#define ledPin D5
#define buzzer D6

void setup() {
  Serial.begin(9600);

  pinMode(flamePin, INPUT);
  pinMode(ledPin, OUTPUT);
  pinMode(buzzer, OUTPUT);

  Blynk.begin(BLYNK_AUTH_TOKEN, ssid, pass);
}

void loop() {
  Blynk.run();

  int flame = digitalRead(flamePin);
  int gas = analogRead(mqPin);

  Blynk.virtualWrite(V0, flame);
  Blynk.virtualWrite(V1, gas);

  if(flame == 0 || gas > 400){
    digitalWrite(ledPin, HIGH);
    digitalWrite(buzzer, HIGH);
  }
  else{
    digitalWrite(ledPin, LOW);
    digitalWrite(buzzer, LOW);
  }

  delay(1000);
}
